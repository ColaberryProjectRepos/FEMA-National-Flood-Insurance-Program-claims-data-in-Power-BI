1.  Create measures in Claims table:  one measure at a time until all 9 measures are created by following the example.


Copy and paste:


.Claims Made = COUNT(Claims[PolicyCount])

.Claims made % = DIVIDE(  COUNT('Claims'[PolicyCount]),  COUNTX(ALL('Claims'), 'Claims'[PolicyCount]) )

.Claims Made for KPI = IF( ISFILTERED(Claims[State]), DIVIDE([.Claims Made],DISTINCTCOUNT('Calendar'[Year])), IF( ISFILTERED(Claims) || ISFILTERED(Disaster), [.Claims Made], DIVIDE([.Claims Made],DISTINCTCOUNT('Calendar'[Year])) ) )

.Claims Made YoY = VAR __PREV_YEAR =  CALCULATE(  COUNT('Claims'[PolicyCount]),  DATEADD('Calendar'[Date], -1, YEAR)  ) RETURN  DIVIDE(  COUNT('Claims'[PolicyCount]) - __PREV_YEAR,  __PREV_YEAR  )

.Claims Made YoY for KPI = SWITCH (  TRUE (),  [.Claims Made YoY] < 0, UNICHAR ( 9660 ),  UNICHAR ( 9650 ) ) & ROUND ( [.Claims Made YoY], 2 ) * 100 & " %"

.Claims Paid = CALCULATE(COUNT('Claims'[PolicyCount]), 'Claims'[TotalAmountPaid] > 0)

.Cost per Claim = DIVIDE( SUM('Claims'[TotalAmountPaid]), [.Claims Paid])

.Cost per Claim YoY = VAR __PREV_YEAR = CALCULATE([.Cost per Claim], DATEADD('Calendar'[Date], -1, YEAR)) RETURN  DIVIDE([.Cost per Claim] - __PREV_YEAR, __PREV_YEAR)

.Cost per Claim YoY for KPI = SWITCH (  TRUE (),  [.Cost per Claim YoY] < 0, UNICHAR ( 9660 ),  UNICHAR ( 9650 ) ) & ROUND ( [.Cost per Claim YoY], 2 ) * 100 & " %"
